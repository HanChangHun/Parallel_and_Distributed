    int chunk_size = arr_size / num_th;

        start2 = t * chunk_size;
        if (t != num_th - 1)
            end2 = start2 + chunk_size - 1;

            end2 = arr_size - 1;
        td_arr[t].arr = arr;
        td_arr[t].start = start2;
        td_arr[t].end = end2;
